# Public origin the app is served from (used for strict CSRF origin checks and
# secure-cookie behaviour). Set to your Zoraxy-fronted https URL in production.
APP_ORIGIN=https://wiim.example.com

# 32+ byte random secret. Generate with: openssl rand -base64 48
# Used to sign session/CSRF artifacts. Changing it invalidates all sessions.
AUTH_SECRET=change-me-openssl-rand-base64-48

# Where the SQLite database + runtime data live (mounted as a Docker volume).
DATA_DIR=/data

# The app runs behind Zoraxy (a trusted reverse proxy), so honour
# X-Forwarded-For / X-Forwarded-Proto for client IP + https detection.
TRUST_PROXY=true

# Listen port (Zoraxy points here).
PORT=3000

# Session cookies are Secure (https-only) by default in production. Set to
# false ONLY when testing over plain http on a LAN IP. Behind https, leave unset.
# COOKIE_SECURE=false

# --- Cloudflare Turnstile (bot protection on login) -------------------------
# You can also set these later in Settings. If left blank, Turnstile is
# disabled until configured (login still requires a password).
TURNSTILE_SITE_KEY=
TURNSTILE_SECRET_KEY=
